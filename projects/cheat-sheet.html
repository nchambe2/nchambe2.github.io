sql cheatsheet

create

create database name_of_database;

create table table_name (

id integer primary key,

column_name data_type);

insert

insert into table_name

values (value_1, value_2, value_3);

insert into table_name(column_name, column_name, column_name)

values (value_1, value_2, value_3);

select

select * from table_name;

select column_name from table_name;

select column_name, column_name from table_name;

where

select column_name from table_name

where condition = value;

select column_name from table_name

where condition = value and condition = value;

select column_name from table_name

where condition = value or condition = value;

order

select column_name from table_name

order by column_name;

select column_name from table_name

order by column_name asc;

update

update table_name

set column_name = value

where column_name = value;

delete

delete from table_name

where column = value;

drop

drop table table_name;

alter

alter table table_name

addcolumn_name data_type;

alter table table_name

drop column column_name;

inner join

select column_name from table_name

inner join other_table_name

on table_name.column_name=other_table_name.column_name;

left join

select column_name from table_name

left join other_table_name

on table_name.column_name=other_table_name.column_name;

right join

select column_name from table_name

right join other_table_name

on table_name.column_name=other_table_name.column_name;
